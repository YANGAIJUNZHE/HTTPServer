08. 解析指令并返回

代码为：

static int32_t handle_request(
    const uint8_t *req, uint32_t reqlen, uint8_t *rep, uint32_t *replen) {
    char *cmds[10];
    int len = parse_request(req, reqlen, cmds);

    if (len == 3) {
        kv_set(cmds[1], cmds[2]);
        const char *msg = "OK";
        strcpy((char *) rep, msg);
        *replen = strlen(msg);
    } else if (len == 2) {
        char *value = kv_get(cmds[1]);
        char *msg = value ? value : "Key not found!";
        strcpy((char *) rep, msg);
        *replen = strlen(msg);
    }
}

解析并返回代码：

/**
 * 消息分为两部分：消息头(4 byte)，消息体(由消息头的长度指定)
 * 消息格式：
 * 消息1长度|消息1|消息2长度|消息2|等等...(|只做例子分割，不存在消息中)
 */
uint32_t process_on_request(int conn_fd) {
    // 4 byte header, string array +1
    char r_buff[4 + MAX_MESSAGE_LEN + 1];

    // 读取消息头
    unsigned len = read_message_header(conn_fd, r_buff);
    if (len < 1) return -1;

    // 读取消息体
    unsigned err = read_message_body(conn_fd, &r_buff[4], len);
    if (err) return err;

    char w_buf[4 + MAX_MESSAGE_LEN];
    uint32_t wlen = 0;
    err = handle_request(&r_buff[4], len, &w_buf[4], &wlen);

    memcpy(w_buf, &wlen, 4);

    return write_message(conn_fd, w_buf, 4 + len);
}
