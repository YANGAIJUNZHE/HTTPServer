09. 存储KV

代码为：

/**
 * 一个非常低效率的KV实现，建议使用hash实现
 */
#include <kv_store.h>
#include <stdio.h>
#include <string.h>

#define MAX_KV_CAPACITY 1000  // 最大容量

struct KeyValue kv_store[MAX_KV_CAPACITY];
int current_size = 0; // 当前store大小

void kv_set(char *key, char *value) {
    // Check if the key already exists
    for (int i = 0; i < current_size; ++i) {
        if (strcmp(kv_store[i].key, key) == 0) {
            // Key found, update the value
            strcpy(kv_store[i].value, value);
            return;
        }
    }

    // Key not found, add a new entry if space is available
    if (current_size < MAX_KV_CAPACITY) {
        strcpy(kv_store[current_size].key, key);
        strcpy(kv_store[current_size].value, value);
        current_size++;
    } else {
        printf("Key-value store is full!\n");
    }
}

char *kv_get(char *key) {
    for (int i = 0; i < current_size; ++i) {
        if (strcmp(kv_store[i].key, key) == 0) {
            // Key found, return the value
            return kv_store[i].value;
        }
    }
    // Key not found
    return NULL;
}

void kv_delete(char *key) {
    int found = 0;

    for (int i = 0; i < current_size; ++i) {
        if (strcmp(kv_store[i].key, key) == 0) {
            // Shift elements left to fill the gap
            for (int j = i; j < current_size - 1; ++j) {
                strcpy(kv_store[j].key, kv_store[j + 1].key);
                strcpy(kv_store[j].value, kv_store[j + 1].value);
            }
            // Clear the last element
            strcpy(kv_store[current_size - 1].key, "");
            strcpy(kv_store[current_size - 1].value, "");
            current_size--;
            found = 1;
            break;
        }
    }

    if (!found) {
        printf("Key '%s' not found.\n", key);
    }
}
